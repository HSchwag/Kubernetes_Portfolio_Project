apiVersion: apps/v1
kind: Deployment
metadata:
  name: kube-center
spec:
  replicas: 2
  selector:
    matchLabels:
      app: kube-center
  template:
    metadata:
      labels:
        app: kube-center
    spec:
      containers:
        - name: django
          image: kube-center:v2
          imagePullPolicy: Never
          ports:
            - containerPort: 8000
          readinessProbe:
            httpGet:
              path: /healthz/
              port: 8000
            initialDelaySeconds: 5
          livenessProbe:
            httpGet:
              path: /healthz/
              port: 8000
            initialDelaySeconds: 10
            periodSeconds: 5
---
apiVersion: v1
kind: Service
metadata:
  name: kube-center-svc
spec:
  selector:
    app: kube-center
  ports:
    - port: 80
      targetPort: 8000
  type: NodePort
